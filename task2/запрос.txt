SELECT
    Статья.Заголовок,
    Комментарии.Текст AS Комментарий,
    Оценки.Значение AS Оценка
FROM
    Статья
JOIN
    Статьи_Авторы ON Статья.ID_Статьи = Статьи_Авторы.ID_Статьи
JOIN
    Пользователи ON Статьи_Авторы.ID_Пользователя = Пользователи.ID_Пользователя
LEFT JOIN
    Комментарии ON Статья.ID_Статьи = Комментарии.ID_Статьи
LEFT JOIN
    Оценки ON Статья.ID_Статьи = Оценки.ID_Статьи
WHERE
    Пользователи.Никнейм = 'Никнейм_автора' OR Пользователи.ФИО = 'ФИО_автора';
